   100→const fmtTime = (iso: string | null): string => {
   101→  if (!iso) return "";
   102→  const d = new Date(iso);
   103→  return d.toLocaleString("en-US", {
   104→    month: "short",
   105→    day: "numeric",
   106→    hour: "numeric",
   107→    minute: "2-digit",
   108→  });
   109→};
   110→
   111→const fmtRelative = (iso: string | null): string => {
   112→  if (!iso) return "just now";
   113→  const diff = Date.now() - new Date(iso).getTime();
   114→  const min = Math.floor(diff / 60000);
   115→  if (min < 1) return "just now";
   116→  if (min < 60) return `${min}m ago`;
   117→  const hr = Math.floor(min / 60);
   118→  if (hr < 24) return `${hr}h ago`;
   119→  const day = Math.floor(hr / 24);
   120→  return `${day}d ago`;
   121→};

   557→                          <Button
   558→                            size="sm"
   559→                            variant="ghost"
   560→                            onClick={() => onNavigate("ai-agent", a.ticker)}
   561→                            className="h-6 text-[10px] text-slate-400 hover:text-amber-300 hover:bg-slate-800 px-2"
   562→                          >
   563→                            <Bot className="h-3 w-3 mr-1" />
   564→                            Analyze with AI Agent
   565→                          </Button>
   566→                        </div>
   567→                      </div>
   568→                    );
   569→                  })}
   570→                  {offset + LIMIT < total && (
   571→                    <Button
   572→                      variant="outline"
   573→                      size="sm"
   574→                      onClick={() => setOffset((o) => o + LIMIT)}
   575→                      className="w-full border-slate-700 text-slate-300 hover:bg-slate-800"
   576→                    >
   577→                      Load more ({total - offset - LIMIT} remaining)
   578→                    </Button>
   579→                  )}
   580→                </>
   581→              )}
   582→            </CardContent>
   583→          </Card>
   584→        </div>
   585→
   586→        {/* Right sidebar: trending tickers */}
   587→        <div className="space-y-4">
   588→          <Card className="border-slate-700 bg-slate-800/40">
   589→            <CardHeader className="pb-2">
   590→              <CardTitle className="text-sm font-semibold flex items-center gap-2 text-slate-100">
   591→                <TrendingUp className="h-4 w-4 text-emerald-400" />
   592→                Trending Tickers (24h)
   593→              </CardTitle>
   594→              <CardDescription className="text-xs text-slate-400">
   595→                Top 10 by mention count
   596→              </CardDescription>
   597→            </CardHeader>
   598→            <CardContent className="space-y-1.5 max-h-[400px] overflow-y-auto aib-scroll">
   599→              {trending.length === 0 ? (
   600→                <div className="text-xs text-slate-500 py-4 text-center">
   601→                  No ticker data yet. Run a scan.
   602→                </div>
   603→              ) : (
   604→                trending.map((t) => {
   605→                  const up = t.avgScore >= 0;
   606→                  const pct = Math.min(100, Math.abs(t.avgScore) * 100);
   607→                  return (
   608→                    <button
   609→                      key={t.ticker}
   610→                      onClick={() => {
   611→                        setTickerInput(t.ticker);
   612→                        setTickerFilter(t.ticker);
   613→                        setOffset(0);
   614→                      }}
   615→                      className="w-full text-left rounded-md border border-slate-700/60 bg-slate-900/40 px-2 py-1.5 hover:border-slate-600"
   616→                    >
   617→                      <div className="flex items-center justify-between text-xs">
   618→                        <span className="font-bold text-amber-300">
   619→                          ${t.ticker}
   620→                        </span>
   621→                        <span className="text-[10px] text-slate-500">
   622→                          {t.mentions} mentions
   623→                        </span>
   624→                      </div>
   625→                      <div className="mt-1 flex items-center gap-2">
   626→                        <div className="flex-1 h-1.5 bg-slate-700 rounded-full overflow-hidden">
   627→                          <div
   628→                            className={cn(
   629→                              "h-full",
   630→                              up ? "bg-emerald-500" : "bg-rose-500"
   631→                            )}
   632→                            style={{ width: `${pct}%` }}
   633→                          />
   634→                        </div>
   635→                        <span
   636→                          className={cn(
   637→                            "font-mono tabular-nums text-[10px] w-12 text-right",
   638→                            up ? "text-emerald-400" : "text-rose-400"
   639→                          )}
   640→                        >
   641→                          {up ? "+" : ""}
   642→                          {t.avgScore.toFixed(2)}
   643→                        </span>
   644→                      </div>
   645→                      <div className="mt-0.5 flex items-center gap-2 text-[9px] text-slate-500">
   646→                        <span className="text-emerald-500">↑{t.bullish}</span>
   647→                        <span className="text-rose-500">↓{t.bearish}</span>
   648→                      </div>
   649→                    </button>
   650→                  );
   651→                })
   652→              )}
   653→            </CardContent>
   654→          </Card>
   655→        </div>
   656→      </div>
   657→
   658→      {/* Sentiment trend chart */}
   659→      <Card className="border-slate-700 bg-slate-800/40">
   660→        <CardHeader className="pb-2">
   661→          <CardTitle className="text-sm font-semibold flex items-center gap-2 text-slate-100">
   662→            <Activity className="h-4 w-4 text-cyan-400" />
   663→            Sentiment Trend (7 days · hourly buckets)
   664→          </CardTitle>
   665→        </CardHeader>
   666→        <CardContent>
   667→          {trend.length === 0 ? (
   668→            <div className="text-xs text-slate-500 py-8 text-center">
   669→              No trend data yet — scans populate this chart over time.
   670→            </div>
   671→          ) : (
   672→            <div className="h-[220px] w-full">
   673→              <ResponsiveContainer width="100%" height="100%">
   674→                <AreaChart data={trend}>
   675→                  <defs>
   676→                    <linearGradient id="sentArea" x1="0" y1="0" x2="0" y2="1">
   677→                      <stop offset="0%" stopColor="#22d3ee" stopOpacity={0.4} />
   678→                      <stop offset="100%" stopColor="#22d3ee" stopOpacity={0} />
   679→                    </linearGradient>
   680→                  </defs>
   681→                  <XAxis
   682→                    dataKey="ts"
   683→                    tick={{ fill: "#64748b", fontSize: 10 }}
   684→                    tickFormatter={(v) =>
   685→                      new Date(v).toLocaleDateString("en-US", {
   686→                        month: "short",
   687→                        day: "numeric",
   688→                      })
   689→                    }
   690→                    minTickGap={40}
   691→                  />
   692→                  <YAxis
   693→                    domain={[-1, 1]}
   694→                    tick={{ fill: "#64748b", fontSize: 10 }}
   695→                    tickFormatter={(v) => v.toFixed(1)}
   696→                  />
   697→                  <RTooltip
   698→                    contentStyle={{
   699→                      background: "#0f172a",
   700→                      border: "1px solid #334155",
   701→                      borderRadius: "8px",
   702→                      color: "#f1f5f9",
   703→                      fontSize: "12px",
   704→                    }}
   705→                    labelFormatter={(v) => new Date(v).toLocaleString("en-US")}
   706→                    formatter={(value: number) => [value.toFixed(3), "Sentiment"]}
   707→                  />
   708→                  <ReferenceLine y={0} stroke="#475569" strokeDasharray="3 3" />
   709→                  <Area
   710→                    type="monotone"
   711→                    dataKey="score"
   712→                    stroke="#22d3ee"
   713→                    strokeWidth={1.5}
   714→                    fill="url(#sentArea)"
   715→                    isAnimationActive={false}
   716→                  />
   717→                </AreaChart>
   718→              </ResponsiveContainer>
   719→            </div>
   720→          )}
   721→        </CardContent>
   722→      </Card>
   723→    </div>
   724→  );
   725→}
